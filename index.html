Core Methods

This is a set of methods available at the core of Stress.

include external JavaScript files. These are files that are part of MVC library and should be in the /assets/js/framework folder.
__.include({
    model        : 'ModelName'    // file name 'model.ModelName.js'
    , controller : 'ControllName' // file name: 'controller.ControllerName.js'
    , view       : 'ViewName'     // file name: 'view.ViewName.js'
});

<script src='/assets/js/framework/models/model.ModeName.js' type='text/javascript'></script>
<script src='/assets/js/framework/controllers/controller.ControllerName.js' type='text/javascript'></script>
<script src='/assets/js/framework/views/view.ViewName.js' type='text/javascript'></script>

To include multiple libraries, use Arrays
__.include({
    model        : ['ModelName', 'OtherModel']         // file name 'model.ModelName.js'
    , controller : ['ControllName', 'OtherController'] // file name: 'controller.ControllerName.js'
    , view       : ['ViewName', 'OtherView']           // file name: 'view.ViewName.js'
});

<script src='/assets/js/framework/models/model.OtherModel.js' type='text/javascript'></script>

Collections

The core of the the Stress framework is Collections. Collections allow the view to work with data models and manipulate them when needed.

Set a Collection
__.model.Collection.set({
    name : "MyCollection",
    model : {
        MyArray   : [1,2,3],
        MyBoolean : true,
        MyNull     : null,
        MyNumber   : 123,
        MyObject   : {
            "a" : "b",
            "c" : "d",
            "e" : "f"
        },
        MyString : "Hello World"
    }
});
Get a Collection
// The entire Collection
//     @CollectionName
var MyCollection = __.model.Collection.get("MyCollection");

// A Property from a Collection
// Will return only the MyObject property within the MyCollection Collection
//     @CollectionName.PropertyName
var MyCollection = __.model.Collection.get("MyCollection.MyObject");
Size of a Collection
// Returns the size (length) of a Collection
//     @CollectionName
var MyCollectionSize = __.model.Collection.size("MyCollection");
insertInto a Collection
// Insert a new key into a Collection. Used when a Collection is a single-dimensional array and not an object
//     @CollectionName
//     @value
__.model.Collection.insetInto("MyCollection", 12345);

// Insert a new Key/Value pair into a Collection. Used when a Collection is a multi-dimensional array/object
//     @CollectionName
//     @key
//     @value
__.model.Collection.insertInto("MyCollection", {
    key   : 'MyNewKey',
    value : [1,2,3,4]
});
Update a Collection
// Updates a key/property within a Collection
//     @CollectionName
//     @where
//     @KeyValuePair
__.model.Collection.update(
    "MyCollection",                         // Name of the Collection
    { key : "theKey", equals : 25 },        // Search criteria (where clause)
    { NameOfPoperty : "New Value for Key" } // Name of key to update and its value
);
deleteFrom a Collection
// Deletes a key or property from the Collection
//
// the Collection is an single dimension array/object
// specify the Collection and the array key or property
//     @CollectionName
//     @key
__.model.schema.deleteFrom({
    "MyCollection",    // Name of Collection
    key : [23|"keyname"] // key to remove
});

// or if the you need to remove the key from an array within a collection

search a Collection
// Searches a collection for a given value
// Can compare against another object if one is specified
//     {
//          @Collection : "CollectionName"
//          @where : { 
//              key : "NameofCollectionKey", 
//              equals : "ValueOfKey" -or- Array,
//              requires : "all|any",
//              ignoreKeys : Array() 
//          }
//          @notIn : { 
//              list : Array(), 
//              CollectionKey : "NameOfKeyInCollection", 
//              ListKey : "IfListIsObjectSpecifyKey" 
//          }
//      }

__.model.Collection.set({
    name : "MyCollection",
    model : Array(
        { Id : 0, Title : "A Title" }, 
        { Id : 1, Title : "Another Object"}, 
        { Id : 2, Title : "This is cool"}, 
        { Id : 3, Title : "Another Test Object" } 
    )
});

ComparedAgainst  = Array(0,2,3,4,5);

__.model.Collection.search({
    Collection : "MyCollection", 
    Where      : { key : "Title", equals : "Another"}, 
    NotIn      : { list : ComparedAgainst, CollectionKey : "Id" } 
    RemoveStopWords : true|false
});
// @returns 1 records because the Id for 'Another Test Object' is in the ComparedAgainst array


//
// Evals property can be used to when you want to check for <, <=, =>, >, ==
//
__.model.Collection.search({
    Collection : "MyCollection", 
    Evals      : Array(
        { Key : 'Id', Operator : '>=', Value : 2 }
    )
);
// @returns 2 records Array({Id : 2, Title : 'This is cool' }, { Id : 3, Title : 'Another Test Object' });

toDOM method. Using a Collection to create a DOM object and output to HTML.

The DOM Element that is populated by the Collection: The LI elements are created using the 'MyModel' Collection.

    
//
// __.dom.toDOM()
// reads a DOM object and adds it to the model's Collection
//
// @selector : [object][required]
// jQuery pointer for DOM object
//
// @ModelName : [string][required]
// name of the model you want to store the DOM object
//
// @TagName : [string][optional|required if no]
//
// @GetDOMObject : [bool][optional]
// whether or not to store the DOM reference in the model
//
// @bindings : [array][optional]
// specified events binds tied to element created from the model
//
// @IterationCallback : [function][optional]
// overrides the built-in creation of each element. this allows you to write your own function for building your HTML based on the output from the specified model. 
// NOTE: IterationCallback will override the bindings, GetDOMObject, and SetAttributes settings
//
// @callback : [function][optional]
// callback function to be fired after the DOM object has been stored in the Collection
//
__.model.toDOM({
    selector       : $('#OtherUL'),
    ModelName      : 'MyModel',
    TagName        : 'li',
    GetDOMObject   : false,
    SetAttributes  : [
        'data-itemid',
        'class'
    ],
    bindings : [
        { 'click': function(event) { alert($(event.target).text()); } }
    ],
    IterationCallback : function(event, Item, Index) {
        $('#OtherUL').append("<li>" + Item.html + "</li>");
    },
    callback : function(event) {
        $('#OtherUL> li').on('click', function() { 
            alert($(this).text());
        });
    }
});
